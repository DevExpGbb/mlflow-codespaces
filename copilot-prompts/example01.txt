# create a sql query to get the number of rows in a table
import sqlite3

# connect to the database
conn = sqlite3.connect("newnum.db")
cursor = conn.cursor()

# create a table named numbers
cursor.execute("CREATE TABLE numbers(num int)")

# create a query
query = "SELECT COUNT(*) FROM numbers"

# execute the query
cursor.execute(query)

# fetch the result
result = cursor.fetchone()

print("The number of rows in the table is: ", result[0])

# close the connection
conn.close()
